\begin{algorithm}
\caption{\textsc{Bellman-Ford}($G, w, s$)}
\begin{algorithmic}
\PROCEDURE{Initialize-Single-Source}{$G, s$}\ENDPROCEDURE
\FOR{$i=1$ to $|G.v| - 1$}
  \FOR{each edge $(u, v)\in G.E$}
    \PROCEDURE{RELAX}{$u, v, w$}\ENDPROCEDURE
  \ENDFOR
\ENDFOR
\FOR{each edge $(u, v)\in G.E$}
  \IF{$v.d > u.d + w(u, v)$}
    \RETURN \textsc{false}
  \ENDIF
\ENDFOR
\RETURN \textsc{true}
\end{algorithmic}
\end{algorithm}
